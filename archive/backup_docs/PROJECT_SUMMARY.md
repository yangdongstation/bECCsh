# 纯粹的美：仅用字符串处理实现数学
for ((i = 1; i <= max_len; i++)); do
    local digit_a="${a: -$i:1}"
    local digit_b="${b: -$i:1}"
    local sum=$((digit_a + digit_b + carry))
    result="$((sum % 10))${result}"
    carry=$((sum / 10))
done
